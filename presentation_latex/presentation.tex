% Slide 3: Proposed Algorithms
\begin{frame}
    \frametitle{Proposed Algorithms for Comparison}
    Aim: allow users to choose between two distinct algorithmic approaches~\cite{8308293}:
    \vspace{0.5em}
    \begin{block}{Algorithm A: Weighted MAUT}
        \begin{itemize}
            \item Linear combination of normalized scores~\cite{schmitt2002maut}.
            \item 100\% transparent; users see exact contributions.
        \end{itemize}
    \end{block}
    \begin{block}{Algorithm B: Constraint-Based / Critiquing Hybrid}
        \begin{itemize}
            \item Hard limits (e.g., ``Max 150\euro{}'') + ``More like this'' critiques~\cite{BORRAS20147370}.
            \item High transparency through explanation of satisfied constraints.
        \end{itemize}
    \end{block}
\end{frame}

% Slide: Theoretical Foundation of Algorithm A
\begin{frame}
    \frametitle{Algorithms Deep Dive}
    
        \textbf{Algorithm A: Multi-Attribute Utility Theory (MAUT)}
            \begin{itemize}
                \item quantifies personal utility for an outcome~\cite{schmitt2002maut}.
                \item Importance weights specify tradeoffs between competing attributes~\cite{schmitt2002maut}.
                \item Overall utility $U(X)$ is calculated using an additive value function: $U(X) = \sum_{i=1}^{5} w_i \cdot u_i(x_i)$~\cite{schmitt2002maut}.
            \end{itemize}
        \textbf{Algorithm B: Constraint-Based \& Critiquing}
            \begin{itemize}
                \item Express desired properties of an``ideal result'' via constraints~\cite{schmitt2002maut}.
                \item Unlike MAUT, can exclude items that do not fit conditions (e.g., price limit)~\cite{schmitt2002maut}.
                \item returns logically sound results, because of user input (no guessing required)~\cite{BORRAS20147370}.
                \item Easy explicable and transparent for users~\cite{schmitt2002maut, BORRAS20147370}.
            \end{itemize}
    
\end{frame}
