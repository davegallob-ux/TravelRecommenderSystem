\begin{frame}
    \frametitle{Project Overview \& Motivation}
    \begin{itemize}
        \item \textbf{The Recommender ``Black-Box'' Problem:} Traditional recommender systems hide their logic, leading users left unknown.~\cite{BORRAS20147370}.
        \item \textbf{Core Goal:} Give users information and control over the ranking logic and filtering mechanisms~\cite{Sarkar2023}.
        \item \textbf{Regulatory Motivation: EU Digital Services Act (DSA):}
            \begin{itemize}
                \item \textit{Article 27 (Recommender System Transparency):} Online platforms must explain the main parameters used in recommender systems and provide user-adjustable options~\cite{eudsa2022}.
                \item \textit{Article 38 (Profiling Options):} Users must be offered at least one option that is not based on profiling~\cite{eudsa2022}.
            \end{itemize}
    \end{itemize}
\end{frame}

\begin{frame}
    \frametitle{Dual-Algorithm System}
    The system implements two algorithmic modes that users can toggle~\cite{8308293}:
    \vspace{0.5em}
    \begin{columns}
        \begin{column}{0.55\textwidth}
            \begin{block}{Algorithm A: Weighted MAUT}
                \begin{itemize}
                    \item \textbf{Approach:} Multi-Attribute Utility Theory (Additive)~\cite{schmitt2002maut}.
                    \item \textbf{User Control:} Interactive sliders to define weights ($w_i$).
                    \item \textbf{Output:} 0--100\% match score and full ranking of all hotels.
                \end{itemize}
            \end{block}
            \begin{block}{Algorithm B: Constraint-Guided Filtering}
                \begin{itemize}
                    \item \textbf{Approach:} Hard filtering using user-defined bounds.
                    \item \textbf{User Control:} Sliders for absolute limit thresholds.
                    \item \textbf{Output:} Exact hotels satisfying all limits.
                \end{itemize}
            \end{block}

% Slide 4: Deep Dive: Weighted MAUT
\begin{frame}
    \frametitle{Deep Dive: Weighted MAUT Implementation}
    \vspace{0.8em}
    \begin{columns}
        \begin{column}{0.52\textwidth}
            Calculates utility $U(X)$ for each hotel $X$~\cite{schmitt2002maut}:
            \begin{equation*}
                U(X) = \frac{\sum_{i=1}^{5} w_i \cdot u_i(x_i)}{\sum_{i=1}^{5} w_i} \times 100\%
            \end{equation*}
            \textbf{Walkthrough:}
            \begin{itemize}
                  \item Attribute Utility ($u_i(x_i)$): hotel's raw values (like a price or distance) are normalized between $0$ (worst option) and $1$ (best option).
                  \item Weighting ($w_i$): normalized value is multiplied by slider weight (from $0$ to $100$) and summed up for all weights and utilities.
                  \item Division ($\sum w_i$): Finally, we divide by the sum of all weights.

            \end{itemize}
        \end{column}
        \begin{column}{0.45\textwidth}
            \begin{center}
                \includegraphics[width=\textwidth,keepaspectratio]{screens/AlgoInfo.png}
                \begin{figure}
                    \tiny\caption{Transparency Banner in UI}
                \end{figure}
            \end{center}
        \end{column}
    \end{columns}
\end{frame}

% Slide 5: Deep Dive: Constraint-Guided Filtering
\begin{frame}
    \frametitle{Deep Dive: Constraint-Guided Filtering}
    \vspace{0.8em}
    A hard-limit filtering approach, showing exactly the fitting hotels
    \vspace{0.5em}
    \begin{itemize}
        \item \textbf{Pure Logic Over Complex Math:} Instead of calculating utility scores, system checks each hotel against user-defined limits.
        \item \textbf{Inequality Criteria Checked:}
            \begin{align*}
                \text{Price} &\le \text{Max Price} & \text{Distance} &\le \text{Max Distance} \\
                \text{Stars} &\ge \text{Min Stars} & \text{Rating} &\ge \text{Min Rating} \\
                \text{Public Transport} &\ge \text{Min Transport}
            \end{align*}
        \item \textbf{Highly Predictable Outcome:} Clear binary classification (a hotel is either included or excluded --- no ``black-box'' scores).
        \item \textbf{Actionable Feedback:} When no hotels match the constraints, the system prompts user to ease their thresholds.
    \end{itemize}
\end{frame}

% Slide 8: Detailed Mathematical Breakdown
\begin{frame}
    \frametitle{Detailed Score Breakdown (Full Transparency)}
    \vspace{0.8em}
    To comply with DSA explainability standards, users can inspect the exact calculations on-demand:
    \vspace{0.5em}
    \begin{columns}
        \begin{column}{0.5\textwidth}
            \begin{block}{Dynamic Calculation Steps}
                \begin{enumerate}
                    \item \textbf{Feature Normalization:} Converts raw hotel attributes (e.g. \$180 price) to a normalized value $u_i \in [0, 1]$ relative to the current city.
                    \item \textbf{Scaling by Weights:} Multiplies each $u_i$ by the user's custom weight $w_i \in [0, 100]$.
                    \item \textbf{Aggregation:} Sums the individual weighted scores: $\sum w_i \cdot u_i(x_i)$.
                    \item \textbf{Final Division:} Divides the sum by the total active weights ($\sum w_i$) to yield the clean $0-100\%$ match score.
                \end{enumerate}
            \end{block}
        \end{column}
        \begin{column}{0.48\textwidth}
