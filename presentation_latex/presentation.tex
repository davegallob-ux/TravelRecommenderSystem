% Slide 1: Project Overview & Motivation
\begin{frame}
    \frametitle{Project Overview \& Motivation}
    \begin{itemize}
        \item \textbf{Standard Recommenders are Black Boxes:} Users rarely know why items are recommended, causing a lack of trust~\cite{BORRAS20147370}.
        \item \textbf{Goal:} Give users direct control and clear info on how things are ranked~\cite{Sarkar2023}.
        \item \textbf{Complying with the EU Digital Services Act (DSA):}
            \begin{itemize}
                \item \textit{Article 27 (Transparency):} Platforms must explain major parameters and let users adjust them~\cite{eudsa2022}.
                \item \textit{Article 38 (Profiling Options):} Must offer at least one choice that is profiling-free~\cite{eudsa2022}.
            \end{itemize}
    \end{itemize}
\end{frame}

% Slide 5: Implemented Algorithms
\begin{frame}
    \frametitle{Two Algorithmic Modes}
    Users can toggle between two distinct modes~\cite{8308293}:
    \vspace{0.5em}
    \begin{columns}
        \begin{column}{0.55\textwidth}
            \begin{block}{Algorithm A: Weighted MAUT}
                \begin{itemize}
                    \item \textbf{Approach:} Multi-Attribute Utility Theory (additive utility)~\cite{schmitt2002maut}.
                    \item \textbf{User Control:} Interactive sliders for custom weights ($w_i$).
                    \item \textbf{Output:} 0--100\% match score.
                \end{itemize}
            \end{block}
            \begin{block}{Algorithm B: Constraint Filtering}
                \begin{itemize}
                    \item \textbf{Approach:} Hard limits filter.
                    \item \textbf{User Control:} Sliders set absolute bounds (e.g. max price).
                    \item \textbf{Output:} Only shows hotels that pass all filters.
                \end{itemize}
            \end{block}
        \end{column}
        \begin{column}{0.42\textwidth}
            \begin{center}
                \includegraphics[width=\textwidth,keepaspectratio]{screens/Sliders.png}

% Slide 6: Deep Dive: Weighted MAUT
\begin{frame}
    \frametitle{Deep Dive: Weighted MAUT}
    \vspace{0.8em}
    \begin{columns}
        \begin{column}{0.52\textwidth}
            Calculates utility $U(X)$ for each hotel $X$~\cite{schmitt2002maut}:
            \begin{equation*}
                U(X) = \frac{\sum_{i=1}^{5} w_i \cdot u_i(x_i)}{\sum_{i=1}^{5} w_i} \times 100\%
            \end{equation*}
            
            \textbf{Current Status: Implemented, but delivers wrong results}
            \begin{itemize}
                \item Algorithm delivers wrong results in a few cases
                \item Bugs and crashes in the implementation.
                \item Normalization calculations occasionally overshoot (e.g. $105\%$)
            \end{itemize}
        \end{column}
        \begin{column}{0.45\textwidth}
            \begin{center}
                \includegraphics[width=\textwidth,keepaspectratio]{screens/AlgoInfo.png}
                \begin{figure}
                    \tiny\caption{Explaining MAUT in the UI}
                \end{figure}
            \end{center}
        \end{column}
    \end{columns}
\end{frame}

% Slide 7: Detailed Score Breakdown
\begin{frame}
    \frametitle{Step-by-Step Score Breakdown}
    \vspace{0.8em}
    To satisfy DSA explainability, the UI shows exact steps:
    \vspace{0.5em}
    \begin{columns}
        \begin{column}{0.5\textwidth}
            \begin{block}{How the scores are calculated}
                \begin{enumerate}
                    \item \textbf{Normalization:} Map raw values (like \$180 price) to a score $u_i \in [0, 1]$ based on best/worst options in that city.
                    \item \textbf{Scaling:} Multiply $u_i$ by the user's slider weight $w_i \in [0, 100]$.
                    \item \textbf{Summing up:} Sum them up: $\sum w_i \cdot u_i(x_i)$.
                    \item \textbf{Final division:} Divide by sum of weights ($\sum w_i$).
                \end{enumerate}
            \end{block}
            \vspace{0.2em}
        \end{column}
        \begin{column}{0.48\textwidth}

% Slide 8: Deep Dive: Constraint-Guided Filtering
\begin{frame}
    \frametitle{Deep Dive: Constraint Filtering}
    \vspace{0.8em}
    A hard-limit filtering approach that just shows matching hotels:
    \vspace{0.5em}
    \begin{itemize}
        \item \textbf{Simple logic, no complex math:} check each hotel against the user's limits.
        \item \textbf{Filters applied:}
            \begin{align*}
                \text{Price} &\le \text{Max Price} & \text{Distance} &\le \text{Max Distance} \\
                \text{Stars} &\ge \text{Min Stars} & \text{Rating} &\ge \text{Min Rating} \\
                \text{Public Transport} &\ge \text{Min Transport}
            \end{align*}
        \item \textbf{Totally predictable outcome:} A hotel must match user's needs --- no useless suggestions for the user.
        \item \textbf{Useful feedback:} If nothing matches, the app suggests loosening the limits.
    \end{itemize}
\end{frame}
